Git

  1. git init
  2. git status
  3. git add 파일명 / git add . // 모든파일을 스테이지에 올리는 명령어
  4. git checkout -- 파일명 // 스테이지에서 내리는 명령어
  5. git rm --cached 파일명 // 스테이지에 올라온 파일을 다시 스테이지에서 내리는 명령어
  6. git commit =>vi edit접속됨, 커밋할 메세지 작성 후 esc :wq로 vi editer 빠져나옴
    // git commit -m "version 1.0.0" // 현재상태를 저장해줘.
  7. git config --global user.email "내이메일작성"
  8. git config --global user.name "bonav00" // 후에 commit 해야함 제일 첨음에 커밋할경우
  9. git log / git shortlog // git의 commit log를 보여준다.
  10. git commit -am "커밋메시지작성" // git add 하고 스테이지올린파일 git commit을 동시에 해주는 명령어
  11. git remote add ex) git remote add origin https://github.com/the-won/test.git // 원격저장소에 연결하는 명령어
  12. git push origin master
  13. git pull origin master
  14. git reset HEAD~1 // 실수한 커밋을 되돌리는 명령어
  15. git revert HEAD // 실수한 커밋을 되돌리는데 커밋 명령어는 유지한다. 원격저장소에 푸시까지 한 경우 사용한다.
  16. git branch <브랜치이름> // ex) git branch development
  17. git checkout <브랜치이름> // ex) git branch master 다른 브랜치에서 "마스터브랜치로 이동"
